### ActionButton

This is a not an existing Vanilla component. It can be used to display submitting states for forms or other actions.

<Canvas>
  <Story name="Default">
    {() => {
      const [isLoading, setIsLoading] = useState(false);
      return (
        <>
          <ActionButton
            appearance="positive"
            loading={isLoading}

### Action successful before loading min duration

<Canvas>
  <Story name="Action successful before loading min duration">
    {() => {
      const [isLoading, setIsLoading] = useState(false);
      return (
        <>
          <ActionButton
            appearance="positive"
            loading={isLoading}
            disabled={isLoading}
            onClick={() => {
              setIsLoading(true);
              setTimeout(() => {
                setIsLoading(false);
              }, 200) 
          }}>
            Click me!
        </ActionButton>
      </>
      );
    }}
  </Story>
</Canvas>
